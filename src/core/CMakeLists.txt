# Requirements

include(FetchContent)
FetchContent_Declare(
    stb
    GIT_REPOSITORY https://github.com/nothings/stb.git
    GIT_TAG 31c1ad3
)
FetchContent_MakeAvailable(stb)

# Sources

set(SOURCES
    Version.cpp              # header file generated by target version

    ColorSpace.hpp
    ColorSpace.cpp

    Thread.hpp
    Thread.cpp
    ThreadList.hpp
    ThreadList.cpp

    Selection.hpp
    Selection.cpp
    Picture.hpp
    #Picture.cpp # TODO
)

# Resources
set(LICENSE_FILE ${CMAKE_SOURCE_DIR}/LICENSE)

# Target: version
add_custom_target(version ALL
    COMMAND ${CMAKE_COMMAND}
    -DSOURCE=${CMAKE_CURRENT_SOURCE_DIR}/Version.hpp.in
    -DOUTPUT=${CMAKE_BINARY_DIR}/generate_version/core/Version.hpp
    -DMAJOR=${PROJECT_VERSION_MAJOR}
    -DMINOR=${PROJECT_VERSION_MINOR}
    -DIS_RELEASE_BUILD=$<CONFIG:Release>
    -DLICENSE_FILE=${LICENSE_FILE}
    -P ${CMAKE_SOURCE_DIR}/tools/generate_version.cmake
    COMMENT "Generating Version.hpp"
)

# Target: core depends on version
add_library(core STATIC ${SOURCES})
add_dependencies(core version)
target_include_directories(core PRIVATE ${CMAKE_BINARY_DIR}/generate_version/core ${stb_SOURCE_DIR})

# Export variables
set(core_INCLUDES ${CMAKE_SOURCE_DIR}/src ${CMAKE_BINARY_DIR}/generate_version PARENT_SCOPE)
