# Requirements

find_package(Qt6 REQUIRED COMPONENTS Core Gui LinguistTools)
find_package(TBB REQUIRED)

# Sources

set(SOURCES
    utils/Version.cpp        # header file generated by target version
    utils/I18n.hpp
    utils/I18n.cpp

    colors/ColorSpace.hpp
    colors/ColorSpace.cpp
    colors/Thread.hpp
    colors/Thread.cpp

    files/IFileSystem.hpp
    files/StandardFileSystem.hpp
    files/StandardFileSystem.cpp
    files/FileLocator.hpp
    files/FileLocator.cpp
)

# Resources
set(TRANSLATIONS
    ${CMAKE_SOURCE_DIR}/resources/i18n/creator/de_DE.ts
)
set(LICENSE ${CMAKE_SOURCE_DIR}/LICENSE)

# Target: version
add_custom_target(version ALL
    COMMAND ${CMAKE_COMMAND}
    -DSOURCE=${CMAKE_CURRENT_SOURCE_DIR}/utils/Version.hpp.in
    -DOUTPUT=${CMAKE_BINARY_DIR}/generate_version/creator/utils/Version.hpp
    -DMAJOR=${PROJECT_VERSION_MAJOR}
    -DMINOR=${PROJECT_VERSION_MINOR}
    -DIS_RELEASE_BUILD=$<CONFIG:Release>
    -P ${CMAKE_SOURCE_DIR}/tools/generate_version.cmake
    COMMENT "Generating Version.hpp"
)

# Target: creator depends on version
qt_add_library(creator STATIC ${SOURCES})
add_dependencies(creator version)
target_link_libraries(creator PRIVATE Qt6::Core Qt6::Gui TBB::tbb)
target_include_directories(creator PRIVATE ${CMAKE_BINARY_DIR}/generate_version/creator ${CMAKE_BINARY_DIR}/generate_version/creator/utils)
qt_add_translations(creator RESOURCE_PREFIX "/i18n/creator/" TS_FILES ${TRANSLATIONS}) # to cleanup, add: LUPDATE_OPTIONS -no-obsolete
qt_add_resources(creator "LICENSE" PREFIX "/" BASE ${CMAKE_SOURCE_DIR} FILES ${LICENSE})

# Export variables
set(creator_INCLUDES ${CMAKE_SOURCE_DIR}/src ${CMAKE_BINARY_DIR}/generate_version PARENT_SCOPE)
