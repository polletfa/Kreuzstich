# Requirements

find_package(Qt6 REQUIRED COMPONENTS Core)

# Sources

set(SOURCES
    # C++ file           Header file
    Version.cpp                           # header file generated by target version
    I18n.cpp             I18n.hpp
)

# Target: version
add_custom_target(version ALL
    COMMAND ${CMAKE_COMMAND}
    -DSOURCE=${CMAKE_CURRENT_SOURCE_DIR}/Version.hpp.in
    -DOUTPUT=${CMAKE_BINARY_DIR}/generate_version/common/Version.hpp
    -DMAJOR=${PROJECT_VERSION_MAJOR}
    -DMINOR=${PROJECT_VERSION_MINOR}
    -DIS_RELEASE_BUILD=$<CONFIG:Release>
    -P ${CMAKE_SOURCE_DIR}/tools/generate_version.cmake
    COMMENT "Generating Version.hpp"
)

# Target: common depends on version
qt_add_library(common STATIC ${SOURCES})
target_include_directories(common PRIVATE ${CMAKE_BINARY_DIR}/generate_version/common)
target_link_libraries(common PRIVATE Qt6::Core)
add_dependencies(common version)

# Export variables
set(common_INCLUDES ${CMAKE_SOURCE_DIR}/src ${CMAKE_BINARY_DIR}/generate_version PARENT_SCOPE)
